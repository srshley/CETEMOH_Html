Boucle for classique

La boucle for permet de répéter un bloc de code un nombre déterminé de fois.

JavaScript — Boucle for 
// Syntaxe : for (initialisation; condition; incrément)
for (let i = 0; i < 5; i++) {
  console.log(`Itération ${i}`);
}
// → Itération 0, 1, 2, 3, 4

// Parcourir un tableau
const fruits = ['pomme', 'banane', 'cerise'];
for (let i = 0; i < fruits.length; i++) {
  console.log(`${i}: ${fruits[i]}`);
}
// → 0: pomme, 1: banane, 2: cerise

// Compter à l'envers
for (let i = 10; i >= 0; i--) {
  console.log(i);
}
// → 10, 9, 8, ..., 0

// Pas de 2
for (let i = 0; i <= 10; i += 2) {
  console.log(i);
}
// → 0, 2, 4, 6, 8, 10

// break — sort de la boucle
for (let i = 0; i < 10; i++) {
  if (i === 5) break;
  console.log(i);
}
// → 0, 1, 2, 3, 4

// continue — passe à l'itération suivante
for (let i = 0; i < 5; i++) {
  if (i === 2) continue;
  console.log(i);
}
// → 0, 1, 3, 4
02

while et do-while

La boucle while répète tant qu'une condition est vraie. do-while garantit au moins une exécution.

JavaScript — while et do-while 
// while — condition testée AVANT
let i = 0;
while (i < 5) {
  console.log(i);
  i++;
}
// → 0, 1, 2, 3, 4

// do-while — condition testée APRÈS (au moins 1 exécution)
let j = 0;
do {
  console.log(j);
  j++;
} while (j < 3);
// → 0, 1, 2

// Même avec condition fausse, do s'exécute 1 fois
let k = 10;
do {
  console.log('Exécuté une fois');
} while (k < 5);

// Utilisation pratique : tant que l'utilisateur n'a pas saisi
let saisie;
do {
  // prompt() en navigateur
  saisie = prompt('Entrez "stop" pour arrêter');
} while (saisie !== 'stop');

// Attention : boucle infinie !
// while (true) { console.log('infini'); }
// Toujours avoir une condition de sortie

03

for...of — pour les itérables

for...of parcourt les valeurs d'un itérable (tableaux, chaînes, Map, Set, etc.).

JavaScript — for...of 
// Tableaux
const fruits = ['pomme', 'banane', 'cerise'];
for (const fruit of fruits) {
  console.log(fruit);
}
// → pomme, banane, cerise

// Chaînes de caractères
const mot = 'JavaScript';
for (const lettre of mot) {
  console.log(lettre);
}
// → J, a, v, a, S, c, r, i, p, t

// Avec index (utilisation d'un compteur)
const couleurs = ['rouge', 'vert', 'bleu'];
for (const [index, couleur] of couleurs.entries()) {
  console.log(`${index}: ${couleur}`);
}
// → 0: rouge, 1: vert, 2: bleu

// Set (valeurs uniques)
const ensemble = new Set(['a', 'b', 'a', 'c']);
for (const val of ensemble) {
  console.log(val); // a, b, c (pas de doublon)
}

// for...of ne fonctionne PAS sur les objets
const obj = { nom: 'Alice', age: 25 };
// for (const v of obj) {} // TypeError !
// Pour les objets → for...in ou Object.keys()
💡
for...of vs for classique

for...of est plus lisible que for (let i = 0...). Il n'y a pas d'index à gérer, pas de length à vérifier. Utilisez-le pour tous les itérables sauf quand vous avez besoin de l'index.

04

for...in — pour les objets

for...in parcourt les clés (propriétés) d'un objet.

JavaScript — for...in 
// Parcourir un objet
const personne = {
  nom: 'Alice',
  age: 25,
  ville: 'Paris'
};

for (const cle in personne) {
  console.log(`${cle}: ${personne[cle]}`);
}
// → nom: Alice, age: 25, ville: Paris

// for...in sur un tableau (déconseillé !)
const arr = ['a', 'b', 'c'];
for (const index in arr) {
  console.log(index); // "0", "1", "2" (chaînes !)
}
// Les indices sont des chaînes, pas des nombres
// Préférez for...of pour les tableaux

// Attention : for...in parcourt aussi les propriétés héritées
Array.prototype.custom = 'test';
for (const i in arr) {
  console.log(i); // "0", "1", "2", "custom"
}
// Utilisez hasOwnProperty() pour filtrer
for (const i in arr) {
  if (arr.hasOwnProperty(i)) {
    console.log(i); // seulement "0", "1", "2"
  }
}

Fonctions — function declaration

🖱️ Clic utilisateur → 📡 Événement DOM
📡 Événement DOM → ⚡ addEventListener
⚡ addEventListener → 🎯 Fonction callback
🎯 Fonction callback → ✨ DOM mis à jour

Une fonction est un bloc de code réutilisable. Déclarée avec le mot-clé function.

JavaScript — Déclaration de fonctions 
// Déclaration (hoisting : peut être appelée avant sa définition)
function direBonjour(prenom) {
  return `Bonjour ${prenom} !`;
}

// Appel
console.log(direBonjour('Alice')); // Bonjour Alice !

// Fonction sans paramètre
function heureActuelle() {
  return new Date().toLocaleTimeString();
}
console.log(heureActuelle()); // ex: 14:35:22

// Fonction avec plusieurs paramètres
function addition(a, b) {
  return a + b;
}
console.log(addition(5, 3)); // 8

// Fonction sans return (renvoie undefined)
function afficher(message) {
  console.log(message);
  // pas de return → undefined
}
const resultat = afficher('Test');
console.log(resultat); // undefined

// Fonction avec paramètre par défaut (ES6)
function saluer(prenom = 'Invité') {
  return `Bonjour ${prenom}`;
}
console.log(saluer());       // Bonjour Invité
console.log(saluer('Bob'));  // Bonjour Bob

Fonctions fléchées (arrow functions) =>

Introduites en ES6, les fonctions fléchées sont une syntaxe plus concise et diffèrent sur la gestion de this.

JavaScript — Fonctions fléchées 
// Syntaxe : (paramètres) => expression

// 1 paramètre → parenthèses optionnelles
const carre = x => x * x;
console.log(carre(5)); // 25

// 0 ou plusieurs paramètres → parenthèses obligatoires
const direBonjour = () => 'Bonjour !';
const addition = (a, b) => a + b;

// Corps avec accolades (plusieurs instructions)
const saluer = (prenom) => {
  const msg = `Salut ${prenom}`;
  console.log(msg);
  return msg;
};

// Retour d'objet littéral → parenthèses autour
const creerPersonne = (nom, age) => ({ nom, age });
console.log(creerPersonne('Alice', 25));
// → { nom: "Alice", age: 25 }

// Fonction classique vs fléchée
const classique = function(a, b) {
  return a + b;
};

const flechee = (a, b) => a + b;

// Callback : les fléchées brillent
const nombres = [1, 2, 3, 4];
const doubles = nombres.map(n => n * 2);
console.log(doubles); // [2, 4, 6, 8]

Paramètres & Return — approf.

JavaScript offre des fonctionnalités avancées pour les paramètres et le retour de fonctions.

JavaScript — Paramètres avancés 
// Rest parameters (...) — nombre variable d'arguments
function somme(...nombres) {
  return nombres.reduce((acc, n) => acc + n, 0);
}
console.log(somme(1, 2, 3, 4));  // 10
console.log(somme(5, 10));        // 15

// Destructuration des paramètres
const utilisateur = { nom: 'Alice', age: 25, ville: 'Paris' };
function afficherUtilisateur({ nom, age, ville }) {
  console.log(`${nom}, ${age} ans, ${ville}`);
}
afficherUtilisateur(utilisateur);

// Destructuration tableau
function premierEtDernier([premier, ...reste]) {
  return { premier, dernier: reste.pop() };
}
console.log(premierEtDernier([10, 20, 30, 40]));
// → { premier: 10, dernier: 40 }

// Return précoce (early return) — meilleure pratique
function verifierAge(age) {
  if (age < 0) return 'Âge invalide';
  if (age < 18) return 'Mineur';
  return 'Majeur';
}
console.log(verifierAge(-5)); // Âge invalide
console.log(verifierAge(15)); // Mineur
console.log(verifierAge(25)); // Majeur

Portée (scope) & fermeture (closure)

La portée définit quelles variables sont accessibles à un endroit donné du code. Une closure est une fonction qui se "souvient" des variables de son contexte parent.

JavaScript — Portée et closure 
// ── Portée globale / locale ──
const globale = 'accessible partout';

function test() {
  const locale = 'accessible seulement ici';
  console.log(globale); // OK
  console.log(locale);  // OK
}

console.log(globale);   // OK
// console.log(locale); // ReferenceError !

// ── Portée de bloc ──
if (true) {
  const blocVar = 'dans le bloc';
  var ancienneVar = 'pas de bloc';
}
// console.log(blocVar);   // ReferenceError
console.log(ancienneVar);   // "pas de bloc" — fuite !

// ── Closure (fermeture) ──
function creerCompteur() {
  let compteur = 0;        // variable "capturée"

  return function() {
    compteur++;
    return compteur;
  };
}

const compteurA = creerCompteur();
const compteurB = creerCompteur();

console.log(compteurA()); // 1
console.log(compteurA()); // 2
console.log(compteurA()); // 3
console.log(compteurB()); // 1 (indépendant !)

// ── Closure pratique : factory ──
function creerMultiplicateur(facteur) {
  return function(n) {
    return n * facteur;
  };
}

const double = creerMultiplicateur(2);
const triple = creerMultiplicateur(3);
console.log(double(5)); // 10
console.log(triple(5)); // 15

Bilan & mini-glossaire

Ce que vous maîtrisez

  • Boucle for classique (initialisation, condition, incrément)
  • while et do-while et leurs différences
  • for...of pour parcourir les itérables (tableaux, chaînes)
  • for...in pour parcourir les clés d'objets
  • Déclaration de fonctions avec return et paramètres
  • Fonctions fléchées syntaxe concise
  • Paramètres avancés : rest, destructuration, early return
  • Portée (globale, locale, bloc) et closures (fermetures)

Mini-glossaire

Boucle
Structure répétitive : for, while, do-while. Exécute un bloc tant qu'une condition est vraie.
Itérable
Objet qu'on peut parcourir avec for...of : tableaux, chaînes, Map, Set.
Fonction fléchée
Syntaxe compacte (x) => x * 2. Pas de propre this, pas de hoisting.
Closure (fermeture)
Fonction qui "capture" les variables de son environnement lexical extérieur.
Hoisting (fonction)
Les déclarations function sont remontées en haut de leur portée, contrairement aux fléchées.
Portée (scope)
Zone du code où une variable est accessible. Globale, fonction, ou bloc avec let/const.
Early return
Technique qui consiste à retourner tôt pour éviter les if/else imbriqués et simplifier la logique.
