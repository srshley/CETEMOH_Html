01

JSON.stringify — objet → chaîne JSON

JSON (JavaScript Object Notation) est un format de données textuel léger. JSON.stringify convertit un objet JS en chaîne JSON.

JavaScript — JSON.stringify 
const personne = {
  nom: 'Alice',
  age: 25,
  adresse: {
    ville: 'Paris',
    codePostal: '75001'
  },
  loisirs: ['lecture', 'voyages'],
  estMajeur: true
};

// Convertir en JSON
const json = JSON.stringify(personne);
console.log(json);
// → {"nom":"Alice","age":25,"adresse":{"ville":"Paris","codePostal":"75001"},"loisirs":["lecture","voyages"],"estMajeur":true}

// JSON est une chaîne de caractères
console.log(typeof json); // "string"

// Avec espacement (indentation lisible)
const jsonPretty = JSON.stringify(personne, null, 2);
console.log(jsonPretty);
/* {
  "nom": "Alice",
  "age": 25,
  "adresse": { ... }
} */

// Les méthodes sont ignorées
const avecMethode = {
  nom: 'Bob',
  saluer() { return 'Bonjour'; }
};
console.log(JSON.stringify(avecMethode));
// → {"nom":"Bob"}  (méthode ignorée)

// undefined, Symbol, fonction → ignorés ou null
console.log(JSON.stringify({ a: undefined, b: Symbol(), c: function(){} }));
// → {}
02

JSON.parse — chaîne JSON → objet

JSON.parse convertit une chaîne JSON valide en objet JavaScript.

JavaScript — JSON.parse 
// Chaîne JSON valide
const json = '{"nom":"Alice","age":25,"loisirs":["lecture"]}';

// Convertir en objet JS
const objet = JSON.parse(json);
console.log(objet.nom);    // Alice
console.log(objet.age);    // 25
console.log(objet.loisirs); // ["lecture"]

// Les guillemets DOIVENT être doubles en JSON
// ❌ JSON.parse("{'nom': 'Alice'}") → ERREUR
// ✅ JSON.parse('{"nom": "Alice"}') → OK

// JSON.parse lance une erreur si le format est invalide
try {
  JSON.parse('{mauvais json}');
} catch (e) {
  console.log('JSON invalide :', e.message);
}

// reviver — transformer les valeurs pendant le parsing
const data = '{"date":"2024-06-15","nombre":"42"}';
const avecReviver = JSON.parse(data, (cle, valeur) => {
  if (cle === 'nombre') return Number(valeur);
  return valeur;
});
console.log(avecReviver.nombre); // 42 (nombre)

// Attention : les dates deviennent des chaînes
const avecDate = JSON.parse('{"creeLe":"2024-01-15T10:00:00.000Z"}');
console.log(avecDate.creeLe); // "2024-01-15T10:00:00.000Z" (string)
console.log(new Date(avecDate.creeLe)); // Date object

03

localStorage — stockage persistant

localStorage stocke des données dans le navigateur. Les données persistent après fermeture du navigateur.

JavaScript — localStorage 
// setItem(clé, valeur) — stocker
localStorage.setItem('theme', 'dark');
localStorage.setItem('langue', 'fr');

// getItem(clé) — récupérer (null si inexistant)
const theme = localStorage.getItem('theme');
console.log(theme); // "dark"

const inexistant = localStorage.getItem('couleur');
console.log(inexistant); // null

// length — nombre d'entrées
console.log(localStorage.length); // 2

// key(index) — nom de la clé à l'index donné
console.log(localStorage.key(0)); // "theme"

// key() pour parcourir
for (let i = 0; i < localStorage.length; i++) {
  const cle = localStorage.key(i);
  console.log(`${cle}: ${localStorage.getItem(cle)}`);
}

// Attention : tout est stocké en chaîne !
localStorage.setItem('age', 25);
console.log(typeof localStorage.getItem('age')); // "string"
// Il faut utiliser JSON.parse/stringify
💡
Voir le localStorage dans le navigateur

Ouvrez les outils développeur (F12), onglet Application → Stockage local. Vous pouvez voir, modifier ou supprimer les entrées directement.

04

Stocker tableaux et objets

Le localStorage ne stocke que des chaînes. Pour stocker des structures complexes, on utilise JSON.

JavaScript — Stocker des structures 
// ══ STOCKER un objet ══
const utilisateur = {
  nom: 'Alice',
  age: 25,
  preferences: {
    theme: 'dark',
    notifications: true
  }
};

// Étape 1 : transformer en JSON
const json = JSON.stringify(utilisateur);

// Étape 2 : stocker
localStorage.setItem('utilisateur', json);

// ══ RÉCUPÉRER l'objet ══
const stocke = localStorage.getItem('utilisateur');
if (stocke) {
  const objet = JSON.parse(stocke);
  console.log(objet.nom);        // Alice
  console.log(objet.preferences.theme); // dark
}

// ══ STOCKER un tableau ══
const taches = [
  { id: 1, texte: 'Apprendre JS', fait: false },
  { id: 2, texte: 'Faire les exercices', fait: true }
];

localStorage.setItem('taches', JSON.stringify(taches));

// ══ RÉCUPÉRER le tableau ══
const tachesStockees = JSON.parse(localStorage.getItem('taches') || '[]');
tachesStockees.forEach(t => console.log(t.texte));

// ══ Mettre à jour (lecture → modif → écriture) ══
function ajouterTache(texte) {
  const taches = JSON.parse(localStorage.getItem('taches') || '[]');
  taches.push({ id: Date.now(), texte, fait: false });
  localStorage.setItem('taches', JSON.stringify(taches));
}

// ══ Supprimer ══
function supprimerTache(id) {
  let taches = JSON.parse(localStorage.getItem('taches') || '[]');
  taches = taches.filter(t => t.id !== id);
  localStorage.setItem('taches', JSON.stringify(taches));
}
══ 05 REMOVEITEM & CLEAR ══ -->
05

removeItem et clear

Supprimer des données du localStorage.

JavaScript — Supprimer du localStorage 
// Ajoutons quelques données
localStorage.setItem('test1', 'valeur1');
localStorage.setItem('test2', 'valeur2');
localStorage.setItem('test3', 'valeur3');

// removeItem(clé) — supprime une entrée
localStorage.removeItem('test1');
console.log(localStorage.getItem('test1')); // null

// clear() — supprime TOUT le localStorage
localStorage.clear();
console.log(localStorage.length); // 0

// Vérifier avant d'accéder
const data = localStorage.getItem('maCle');
if (data !== null) {
  console.log(JSON.parse(data));
} else {
  console.log('Clé inexistante');
}

// 👨‍💻 Astuce : préfixer ses clés pour éviter les conflits
localStorage.setItem('monApp_theme', 'dark');
localStorage.setItem('monApp_langue', 'fr');

// Supprimer toutes les clés de l'application
Object.keys(localStorage)
  .filter(cle => cle.startsWith('monApp_'))
  .forEach(cle => localStorage.removeItem(cle));
⚠️
Attention — clear() supprime tout

localStorage.clear() supprime toutes les entrées, y compris celles d'autres sites. Utilisez removeItem pour une suppression ciblée.

sessionStorage — stockage temporaire

sessionStorage est identique à localStorage, mais les données sont effacées à la fermeture de l'onglet.

JavaScript — sessionStorage 
// Même API que localStorage
sessionStorage.setItem('panier', JSON.stringify([
  { produit: 'Clavier', prix: 50 },
  { produit: 'Souris', prix: 30 }
]));

const panier = JSON.parse(sessionStorage.getItem('panier') || '[]');
console.log(panier.length); // 2

// sessionStorage est effacé quand l'onglet est fermé
// Utile pour : panier temporaire, données de formulaire, état de session

// Différences clés :
// localStorage  → 10 Mo, persistant, partagé entre tous les onglets
// sessionStorage → 5-10 Mo, temporaire, isolé par onglet

// Exemple : formulaire en cours
function sauvegarderBrouillon() {
  const formulaire = {
    nom: document.getElementById('nom')?.value || '',
    message: document.getElementById('message')?.value || ''
  };
  sessionStorage.setItem('brouillon', JSON.stringify(formulaire));
}

function restaurerBrouillon() {
  const brouillon = sessionStorage.getItem('brouillon');
  if (brouillon) {
    const data = JSON.parse(brouillon);
    // Restaurer les champs...
  }
}

JavaScript — Cookies (pour info) 
// Lire un cookie
console.log(document.cookie);

// Écrire un cookie
document.cookie = 'theme=dark; path=/; max-age=3600';

// max-age en secondes (3600 = 1 heure)
// Sans max-age, le cookie expire à la fermeture du navigateur

// Pour du stockage côté client :
// ✅ localStorage pour les données persistantes
// ✅ sessionStorage pour les données temporaires (onglet)
// ❌ Cookies réservés à l'authentification côté serveur

// Vérifier la disponibilité
function storageDisponible(type) {
  try {
    const storage = window[type];
    storage.setItem('test', '1');
    storage.removeItem('test');
    return true;
  } catch(e) {
    return false;
  }
}
console.log('localStorage disponible:', storageDisponible('localStorage'));

Exemple complet : Carnet d'adresses

🖱️ Clic utilisateur → 📡 Événement DOM
📡 Événement DOM → ⚡ addEventListener
⚡ addEventListener → 🎯 Fonction callback
🎯 Fonction callback → ✨ DOM mis à jour

Un carnet d'adresses avec stockage localStorage.

JavaScript — Carnet d'adresses 
// === Gestion du carnet avec localStorage ===

const CARNET_CLE = 'monCarnet';

// Obtenir tous les contacts
function getContacts() {
  return JSON.parse(localStorage.getItem(CARNET_CLE) || '[]');
}

// Ajouter un contact
function ajouterContact(nom, email, telephone) {
  const contacts = getContacts();
  contacts.push({
    id: Date.now(),
    nom,
    email,
    telephone,
    creeLe: new Date().toISOString()
  });
  localStorage.setItem(CARNET_CLE, JSON.stringify(contacts));
  return contacts;
}

// Supprimer un contact
function supprimerContact(id) {
  let contacts = getContacts();
  contacts = contacts.filter(c => c.id !== id);
  localStorage.setItem(CARNET_CLE, JSON.stringify(contacts));
  return contacts;
}

// Modifier un contact
function modifierContact(id, nouvellesDonnees) {
  let contacts = getContacts();
  contacts = contacts.map(c =>
    c.id === id ? { ...c, ...nouvellesDonnees } : c
  );
  localStorage.setItem(CARNET_CLE, JSON.stringify(contacts));
  return contacts;
}

// Rechercher
function rechercherContacts(terme) {
  const t = terme.toLowerCase();
  return getContacts().filter(c =>
    c.nom.toLowerCase().includes(t) ||
    c.email.toLowerCase().includes(t)
  );
}

// Afficher les contacts dans le DOM
function afficherContacts() {
  const liste = document.getElementById('liste-contacts');
  if (!liste) return;

  const contacts = getContacts();
  liste.innerHTML = contacts.map(c => `
    

      ${c.nom}
      
${c.email} · ${c.telephone || 'N/A'}

      
    

  `).join('');
}

Bilan & mini-glossaire

Ce que vous maîtrisez

  • JSON.stringify : convertir objet/tableau en chaîne JSON
  • JSON.parse : convertir chaîne JSON en objet/tableau
  • localStorage : setItem, getItem, stockage persistant
  • Stocker des structures avec JSON.stringify/parse
  • removeItem et clear pour supprimer
  • sessionStorage : temporaire, isolé par onglet
  • Différences Cookie vs localStorage vs sessionStorage
  • Exemple carnet d'adresses complet

Mini-glossaire

JSON
JavaScript Object Notation — format de données textuel léger, basé sur la syntaxe des objets JS mais avec guillemets doubles obligatoires.
JSON.stringify
Méthode qui convertit une valeur JS en chaîne JSON. Ignore les fonctions, undefined, Symbol.
JSON.parse
Méthode qui convertit une chaîne JSON valide en valeur JS. Lance une erreur si le format est invalide.
localStorage
Stockage clé/valeur persistant (5-10 Mo). Les données survivent à la fermeture du navigateur.
sessionStorage
Stockage clé/valeur temporaire. Les données sont effacées à la fermeture de l'onglet.
Cookie
Petit fichier texte (4 Ko max) envoyé automatiquement au serveur à chaque requête HTTP.
