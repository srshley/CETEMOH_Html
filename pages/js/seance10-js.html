Modules ES6 — export / import

Les modules permettent de découper le code en fichiers réutilisables. Chaque fichier a sa propre portée.

JavaScript — Export (math.js) 
// ── export nommé ──
export const PI = 3.14159;

export function addition(a, b) {
  return a + b;
}

export function multiplication(a, b) {
  return a * b;
}

// ── export par défaut (un seul par fichier) ──
export default function saluer(nom) {
  return `Bonjour ${nom} !`;
}

// ── export avec alias ──
function interne(n) { return n * 2; }
export { interne as doubler };
JavaScript — Import (app.js) 
// ── Importer dans app.js ──
// Le script HTML doit avoir type="module"
// <script type="module" src="app.js"></script>

// Import nommé
import { PI, addition, multiplication } from './math.js';

console.log(addition(5, 3));       // 8
console.log(PI);                    // 3.14159

// Import par défaut
import saluer from './math.js';
console.log(saluer('Alice'));       // Bonjour Alice !

// Import avec alias
import { addition as add, multiplication as mult } from './math.js';
console.log(add(2, 3)); // 5

// Importer tout
import * as Math from './math.js';
console.log(Math.addition(10, 5)); // 15

// Import dynamique (lazy loading)
const module = await import('./math.js');
console.log(module.PI);
⚠️
Les modules nécessitent un serveur HTTP

Les modules ES6 ne fonctionnent pas en ouvrant le fichier HTML directement (file://). Utilisez Live Server (VS Code) ou un serveur local. Ajoutez type="module" à votre balise script.

02

Class & constructor

Les classes (ES6) sont un sucre syntaxique sur les prototypes. Le constructor initialise les propriétés.

JavaScript — Class 
class Personne {
  constructor(nom, age) {
    this.nom = nom;
    this.age = age;
  }
}

// Instanciation (avec new)
const alice = new Personne('Alice', 25);
const bob = new Personne('Bob', 30);

console.log(alice.nom); // Alice
console.log(bob.age);   // 30
console.log(alice instanceof Personne); // true

// ── Avant les classes (prototypes) ──
function AnciennePersonne(nom, age) {
  this.nom = nom;
  this.age = age;
}
const ancienne = new AnciennePersonne('Charlie', 35);
console.log(ancienne.nom); // Charlie

03

Méthodes de classe

Les méthodes sont définies directement dans le corps de la classe. Pas besoin de function.

JavaScript — Méthodes de classe 
class Personne {
  constructor(nom, age) {
    this.nom = nom;
    this.age = age;
  }

  // Méthode
  saluer() {
    return `Bonjour, je suis ${this.nom}`;
  }

  estMajeur() {
    return this.age >= 18;
  }

  anniversaire() {
    this.age++;
    return `Joyeux anniversaire ${this.nom} ! Vous avez ${this.age} ans.`;
  }

  // Méthode qui utilise une autre méthode
  description() {
    const status = this.estMajeur() ? 'majeur' : 'mineur';
    return `${this.nom}, ${this.age} ans, ${status}`;
  }
}

const alice = new Personne('Alice', 25);
console.log(alice.saluer());        // Bonjour, je suis Alice
console.log(alice.estMajeur());     // true
console.log(alice.anniversaire());  // Joyeux anniversaire Alice ! Vous avez 26 ans.
console.log(alice.description());   // Alice, 26 ans, majeur

// Les méthodes sont sur le prototype (pas sur l'instance)
console.log(Object.hasOwn(alice, 'nom'));        // true
console.log(Object.hasOwn(alice, 'saluer'));     // false (sur le proto)

extends / super — héritage

L'héritage permet à une classe d'hériter des propriétés et méthodes d'une autre classe.

JavaScript — Héritage 
class Personne {
  constructor(nom, age) {
    this.nom = nom;
    this.age = age;
  }

  saluer() {
    return `Bonjour, je suis ${this.nom}`;
  }
}

// extends = hériter de Personne
class Etudiant extends Personne {
  constructor(nom, age, filiere) {
    super(nom, age); // Appelle le constructeur parent
    this.filiere = filiere;
    this.notes = [];
  }

  ajouterNote(note) {
    this.notes.push(note);
  }

  moyenne() {
    if (this.notes.length === 0) return 0;
    const somme = this.notes.reduce((a, n) => a + n, 0);
    return somme / this.notes.length;
  }

  // Surcharger (override) une méthode parent
  saluer() {
    return `${super.saluer()} — Étudiant en ${this.filiere}`;
  }
}

const etudiant = new Etudiant('Alice', 20, 'Informatique');
etudiant.ajouterNote(15);
etudiant.ajouterNote(18);
etudiant.ajouterNote(12);

console.log(etudiant.saluer());  // Bonjour, je suis Alice — Étudiant en Informatique
console.log(etudiant.moyenne()); // 15
console.log(etudiant instanceof Etudiant); // true
console.log(etudiant instanceof Personne);  // true (héritage)

// Classe plus spécialisée
class EtudiantBoursier extends Etudiant {
  constructor(nom, age, filiere, montant) {
    super(nom, age, filiere);
    this.montantBourse = montant;
  }

  infoBourse() {
    return `Bourse de ${this.montantBourse}€/mois`;
  }
}

Classes et manipulation du DOM

🖱️ Clic utilisateur → 📡 Événement DOM
📡 Événement DOM → ⚡ addEventListener
⚡ addEventListener → 🎯 Fonction callback
🎯 Fonction callback → ✨ DOM mis à jour

Les classes sont parfaites pour structurer le code qui manipule le DOM.

JavaScript — Classe + DOM 
class Compteur {
  constructor(selector, initial = 0) {
    this.valeur = initial;
    this.element = document.querySelector(selector);

    if (!this.element) return;

    this.render();
    this.ecouter();
  }

  incrementer() {
    this.valeur++;
    this.render();
  }

  decrementer() {
    this.valeur--;
    this.render();
  }

  render() {
    this.element.innerHTML = `
      
${this.valeur}

      
      
    `;
  }

  ecouter() {
    this.element.addEventListener('click', (e) => {
      if (e.target.classList.contains('btn-plus')) {
        this.incrementer();
      }
      if (e.target.classList.contains('btn-moins')) {
        this.decrementer();
      }
    });
  }
}

// Utilisation :
// const cpt = new Compteur('#app-compteur', 5);

// ── Classe pour gérer une liste ──
class ListeTaches {
  constructor(conteneurId) {
    this.conteneur = document.getElementById(conteneurId);
    this.taches = []; // bientôt avec localStorage
    this.render();
  }

  ajouter(texte) {
    this.taches.push({ id: Date.now(), texte, fait: false });
    this.render();
  }

  supprimer(id) {
    this.taches = this.taches.filter(t => t.id !== id);
    this.render();
  }

  render() {
    if (!this.conteneur) return;
    this.conteneur.innerHTML = this.taches.map(t => `
      

        ${t.texte}
        
      

    `).join('');
  }
}

Méthodes statiques

Les méthodes statiques sont appelées sur la classe elle-même, pas sur les instances.

JavaScript — Static 
class MathUtils {
  static PI = 3.14159;

  static carre(x) {
    return x * x;
  }

  static somme(...nombres) {
    return nombres.reduce((a, n) => a + n, 0);
  }

  static estPair(n) {
    return n % 2 === 0;
  }

  // Factory statique
  static creerPersonne(nom, age) {
    return { nom, age, creeLe: new Date() };
  }

  // Méthode utilitaire de comparaison
  static comparerAge(a, b) {
    return a.age - b.age;
  }
}

// Appel sans instanciation
console.log(MathUtils.PI);             // 3.14159
console.log(MathUtils.carre(5));       // 25
console.log(MathUtils.somme(1, 2, 3)); // 6
console.log(MathUtils.estPair(4));     // true

const personne = MathUtils.creerPersonne('Alice', 25);
console.log(personne); // { nom:"Alice", age:25, creeLe: Date }

// ❌ Impossible d'appeler une méthode statique sur une instance
// const utils = new MathUtils();
// utils.carre(5); // TypeError !

// Exemple concret : config
class Config {
  static get API_URL() {
    return 'https://api.monsite.com';
  }

  static get VERSION() {
    return '1.0.0';
  }
}
console.log(Config.API_URL); // https://api.monsite.com

Getter et setter

Les getters et setters sont des propriétés calculées avec des méthodes get et set.

JavaScript — Getter/Setter 
class Personne {
  constructor(nom, prenom, age) {
    this._nom = nom;
    this._prenom = prenom;
    this._age = age;        // privée par convention (_)
  }

  // Getter — comme une propriété (pas d'appel fonction)
  get nomComplet() {
    return `${this._prenom} ${this._nom}`;
  }

  get age() {
    return this._age;
  }

  // Setter — validation à l'assignation
  set age(valeur) {
    if (valeur < 0 || valeur > 150) {
      throw new Error('Âge invalide');
    }
    this._age = valeur;
  }

  // Getter calculé
  get estMajeur() {
    return this._age >= 18;
  }

  get initiales() {
    return `${this._prenom[0]}${this._nom[0]}`.toUpperCase();
  }
}

const alice = new Personne('Dupont', 'Alice', 25);

// Utilisation comme des propriétés
console.log(alice.nomComplet); // Alice Dupont
console.log(alice.estMajeur);  // true
console.log(alice.initiales);  // AD

// Setter
alice.age = 30; // OK
console.log(alice.age);       // 30
// alice.age = 200; // Erreur !

// Getter avec underscore privé
class Temperature {
  constructor(celsius) {
    this._celsius = celsius;
  }

  get fahrenheit() {
    return this._celsius * 9/5 + 32;
  }

  set fahrenheit(v) {
    this._celsius = (v - 32) * 5/9;
  }

  get celsius() {
    return this._celsius;
  }
}

const temp = new Temperature(25);
console.log(temp.fahrenheit); // 77
temp.fahrenheit = 100;
console.log(temp.celsius);    // ~37.78

Mini-projet : Todo List CRUD + localStorage

Le projet final qui combine classes, DOM et localStorage.

JavaScript — Todo List 
class TodoList {
  constructor(containerId) {
    this.container = document.getElementById(containerId);
    this.taches = this.charger();
    this.render();
  }

  // CRUD
  ajouter(texte) {
    if (!texte.trim()) return;
    this.taches.push({
      id: Date.now(),
      texte: texte.trim(),
      complete: false,
      creeLe: new Date().toISOString()
    });
    this.sauvegarder();
    this.render();
  }

  supprimer(id) {
    this.taches = this.taches.filter(t => t.id !== id);
    this.sauvegarder();
    this.render();
  }

  basculer(id) {
    const tache = this.taches.find(t => t.id === id);
    if (tache) {
      tache.complete = !tache.complete;
      this.sauvegarder();
      this.render();
    }
  }

  // Stockage
  charger() {
    try {
      return JSON.parse(localStorage.getItem('todoList')) || [];
    } catch {
      return [];
    }
  }

  sauvegarder() {
    localStorage.setItem('todoList', JSON.stringify(this.taches));
  }

  // Statistiques
  get total() { return this.taches.length; }
  get faites() { return this.taches.filter(t => t.complete).length; }
  get restantes() { return this.total - this.faites; }

  // HTML
  render() {
    this.container.innerHTML = `
      

        
📋 Ma Todo List

        

          ${this.total} tâches · ${this.faites} faites · ${this.restantes} restantes
        

      

      

        
        
      

      

        ${this.taches.map(t => `
          

            
            ${this.escHtml(t.texte)}
            
          

        `).join('')}
      

    `;
    this.ecouter();
  }

  ecouter() {
    // Ajout
    const form = this.container.querySelector('.todo-form');
    form.addEventListener('submit', (e) => {
      e.preventDefault();
      const input = form.querySelector('input');
      this.ajouter(input.value);
      input.value = '';
    });

    // Délégation pour basculer et supprimer
    this.container.querySelector('.todo-items').addEventListener('click', (e) => {
      const item = e.target.closest('.todo-item');
      if (!item) return;
      const id = Number(item.dataset.id);

      if (e.target.classList.contains('todo-suppr')) {
        this.supprimer(id);
      } else if (e.target.type === 'checkbox') {
        this.basculer(id);
      }
    });
  }

  escHtml(str) {
    const div = document.createElement('div');
    div.textContent = str;
    return div.innerHTML;
  }
}

Glossaire final

Classe
Modèle pour créer des objets. Définie avec class, instanciée avec new. Sucre syntaxique sur les prototypes.
extends
Mot-clé pour créer une classe fille qui hérite d'une classe parente.
super
Mot-clé pour appeler le constructeur ou les méthodes de la classe parente.
Méthode statique
Méthode appelée sur la classe elle-même (MaClasse.methode()), pas sur les instances.
Getter/Setter
Propriétés calculées avec les mots-clés get et set. Permettent la validation et le calcul.
Module ES6
Fichier JS avec son propre scope. Utilise export et import. Nécessite type="module".
CRUD
Create, Read, Update, Delete — les 4 opérations de base de la persistence des données.
Event Loop
Mécanisme de JS qui gère l'ordre d'exécution entre le code synchrone (call stack) et asynchrone (task queue).
Closure
Fonction qui se souvient des variables de son contexte lexical parent, même après la fin de l'exécution du parent.
this
Mot-clé qui fait référence au contexte d'exécution. Sa valeur dépend de comment la fonction est appelée.
